226  void pack(AlpsEncoded& buf) {
227  buf.writeRep(bpar_, endOfBoolParams)
228  .writeRep(ipar_, endOfIntParams)
229  .writeRep(dpar_, endOfDblParams);
230  for (int i = 0; i < endOfStrParams; ++i)
231  buf.writeRep(spar_[i]);
232  for (int i = 0; i < endOfStrArrayParams; ++i) {
233  buf.writeRep(sapar_[i].size());
234  for (size_t j = 0; j < sapar_[i].size(); ++j)
235  buf.writeRep(sapar_[i][j]);
236  }
237  }

240  void unpack(AlpsEncoded& buf) {
241  int dummy;
242  // No need to allocate the arrays, they are of fixed length
243  dummy = static_cast<int>(endOfBoolParams);
244  buf.readRep(bpar_, dummy, false);
245  dummy = static_cast<int>(endOfIntParams);
246  buf.readRep(ipar_, dummy, false);
247  dummy = static_cast<int>(endOfDblParams);
248  buf.readRep(dpar_, dummy, false);
249  for (int i = 0; i < endOfStrParams; ++i)
250  buf.readRep(spar_[i]);
251  for (int i = 0; i < endOfStrArrayParams; ++i) {
252  size_t str_size;
253  buf.readRep(str_size);
254  sapar_[i].reserve(str_size);
255  for (size_t j = 0; j < str_size; ++j){
256  // sapar_[i].unchecked_push_back(std::string());

259  }
